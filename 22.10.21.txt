def f(a,n):
    a = a.split(' ')
    m = len(a)
    a1 = []
    for i in range(1,m):
        if int(a[i]) > n:
            l = a[i]
            a1.append(l)
    return a1
print('Введите строчку чисел через пробел')
a = input()
print('Введите число')
n = int(input())
print(f(a,n))

def glas(a):
    spis = 'а, о, и, е, ё, э, ы, у, ю, я'
    spis = spis.split(', ')
    a = list(a)
    kol = 0
    for i in range(len(a)):
        if a[i] in spis:
            kol += 1
    return kol
    
def sogl(a):
    spis = 'б, в, г, д, ж, з, й, к, л, м, н, п, р, с, т, ф, х, ц, ч, ш, щ'
    spis = spis.split(', ')
    a = list(a)
    kol = 0
    for i in range(len(a)):
        if a[i] in spis:
            kol += 1
    return kol
a = input('Введите строку: ')
fun = int(input('Введите 0, если хотите узнать кол-во гласных русских букв в строке и 1, если согласных: '))
if fun == 0:
    print(glas(a))
elif fun == 1:
    print(sogl(a))
