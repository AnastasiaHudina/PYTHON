def bubble_sort(ls): 
    for i in range(0, len(ls) - 1): 
        for j in range (len(ls) - 1): 
            if (ls[j] > ls[j + 1]): 
                temp = ls[j] 
                ls[j] = ls[j + 1] 
                ls[j + 1] = temp 
    return ls 
 
ls = [10, 5, 1, 16, 3, 50] 
print("Неотсортированный список: ", ls) 
print("Отсортированный список: ", bubble_sort(ls))

list1 = list(map(int, input("Введите список: ").split()))
c = [0] * (max(list1) + 1) 
for item in list1:
    c[item] += 1   
result = [num for num, count in enumerate(c) for i in range(count)] 

print(result)
